#!/bin/sh

# VARS
#########################################################
SCRIPT=$0
TSD="${SNAP_USER_COMMON}/TunerStudioMS"
TSL="https://www.efianalytics.com/TunerStudio/download/TunerStudioMS_v3.1.04.tar.gz"
TS="TunerStudio.tar.gz"
#########################################################

cd ${TSD}

while [ -h "$SCRIPT" ]; do
	SCRIPT=$( readlink "$SCRIPT" )
done

# JAVA PATH
########################################
export JAVA_HOME=$SNAP/usr/lib/jvm/java-1.8.0-openjdk-$SNAP_ARCH
export PATH=$JAVA_HOME/bin:$JAVA_HOME/jre/bin:$PATH
########################################


# SUB-VARS
########################################
#BASE_PATH="${PWD}/${SCRIPT%/*}"
SERIAL_DRIVER="${TSD}/lib/alternateLinuxDrivers/x86_64-linux:"
LAST_PATH=${PWD}
########################################

# GET TunerStudio if it doesn't exist then START

if [ -e "${TSD}" ];

then

  # CLEAN UP, JIC
  rm -rf ${TS}

  java -Dcom.ibm.crypto.provider.DoRSATypeChecking=false -Djava.settings.path=${TSD} -Djava.library.path=${SERIAL_DRIVER}lib -cp ".:lib:plugins" -jar ${TSD}/TunerStudioMS.jar $1
RESULT=$?

else

  # GET TunerStudio
  curl -so ${TS} ${TSL}
  tar xvzf ${TS}
  
  # CLEAN UP, JIC
  rm -rf ${TS}
  
  java -Dcom.ibm.crypto.provider.DoRSATypeChecking=false -Djava.settings.path=${TSD} -Djava.library.path=${SERIAL_DRIVER}lib -cp ".:lib:plugins" -jar ${TSD}/TunerStudioMS.jar $1
RESULT=$?
  
  
fi

if [ -d "$LAST_PATH" ]; then
	cd "$LAST_PATH"
fi

exit $RESULT
